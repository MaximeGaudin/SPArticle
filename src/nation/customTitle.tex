\ifthenelse{\equal{\nationclass}{book}}
{
% Book
	\newcommand{\titlefont}{
	\fontencoding{\encodingdefault}
	\fontfamily{pag}
	\fontseries{bc}
	\fontshape{bf}
	\fontsize{35}{30}
	\selectfont}

	\newcommand{\subtitlefont}{\sl\Large}

	\newlength{\titlehruleheight}
	\setlength{\titlehruleheight}{0.3mm}

	\newlength{\titlespace}
	\setlength{\titlespace}{0.5cm}

	\renewcommand{\maketitle}
	{
		\begin{titlepage}
		\centering
	
		\vspace*{5cm}
		\begin{minipage}{0.8\linewidth}
			\centering
				\hrule height \titlehruleheight
			\vspace*{\titlespace}
	
				{\titlefont\title}~\\
			\vspace*{\titlespace}
	
				\hrule height \titlehruleheight
				~\\{\subtitlefont\subtitle}~\\
			\vspace*{\titlespace}
	
				\textsl{\textbf{Auteurs}\\ \author}
			\vspace*{2cm}
	
			\ifthenelse{\isundefined{\documentlogo}} {}
			{\image{\documentlogo}{0.4}}
		\end{minipage}
		
		\vfill
		{\bf\large \today}
		\end{titlepage}
	}
}
