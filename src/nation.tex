\newcommand{\nHeader}[3]
{
	\title{\textbf{#1}}
	\author{#2}
	\lhead{#3}
	\cfoot{\thepage\ of \pageref{LastPage}}
}

\newcommand{\nPart}[1]
{

			\part{#1}		
		\end{center}
	\end{nVCenter}
	\newpage

	\setcounter{section}{1}
}
