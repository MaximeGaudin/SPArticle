\newcommand{\chapterfont}{
	\fontencoding{\encodingdefault}
	\fontfamily{pag}
	\fontseries{bc}
	\fontshape{bf}
	\fontsize{22}{20}
	\selectfont}

\newcommand{\chaptertitlefont}{
	\fontencoding{\encodingdefault}
	\fontfamily{pag}
	\fontseries{bc}
	\fontshape{bf}
	\fontsize{32}{30}
	\selectfont}


\newcommand{\chapternumfont}{
	\fontencoding{\encodingdefault}
	\fontfamily{pag}
	\fontseries{bc}
	\fontshape{bf}
	\fontsize{72}{70}
	\selectfont
}

\newcounter{chapitre}
\setcounter{chapitre}{1}

\makeatletter
\renewcommand{\chapter}{%
	\cleardoublepage
	\stepcounter{chapitre}
	\thispagestyle{plain}
	\global\@topnum\z@
	\@afterindentfalse
	\secdef\@chapter\@schapter}

\newenvironment{cadrechap}
{
	\begin{list}{}{%
		\setlength{\leftmargin}{0pt}%
		\setlength{\rightmargin}{0cm}%
		\setlength{\itemindent}{0pt}%
		\setlength{\labelsep}{0pt}%
	}
	\item
}
{
	\end{list}
}

% Document
\renewcommand{\@makechapterhead}[1]{% 
	\begin{minipage}[t]{\linewidth}
		\begin{minipage}[t]{0.65\linewidth}
			% Numéro de chapitre
			{\flushleft\minitoc}%
		\end{minipage}
		\hfill
		\begin{minipage}[t][3cm][t]{0.25\linewidth}
			{\chapterfont Chapitre}\\~\\%
			{\chapternumfont\thechapter}%
		\end{minipage}
	\end{minipage}

	% Titre du chapitre
	\vspace*{1cm}
	\begin{center}\chaptertitlefont#1\end{center} 
	\vspace*{2cm}
}

% Table des matières
\renewcommand{\@makeschapterhead}[1]{%   
	\begin{cadrechap}
		\mbox{}\hfill{\bf\huge#1}
	\end{cadrechap}%
	\vspace{2cm}\par
}
\makeatother
