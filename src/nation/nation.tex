%% Extras
% \image[Titre de l'image]{Chemin}{Proportion par rapport à la largeur de la page}
\newcounter{cntFigure}
\setcounter{cntFigure}{1}
\newcommand{\image}[3][null] 
{
\begin{figure}[H]
\centering
	\includegraphics[keepaspectratio=true, width=#3\textwidth]{#2}
	\ifthenelse{\equal{#1}{null}}
	{}
	{
		\caption{#1}
		\label{fig\thecntFigure}
		\stepcounter{cntFigure}
	}
\end{figure}
}

% \wrappedimage[Titre de l'image]{Chemin}{Proportion par rapport à la largeur de la page}{r ou l} 
\newcommand{\wrappedimage}[4][null] 
{
\begin{wrapfigure}{#4}{#3\textwidth}
\centering
	\includegraphics[keepaspectratio=true, width=#3\textwidth]{#2}
	\ifthenelse{\equal{#1}{null}}
	{}
	{
		\caption{#1}
		\label{fig\thecntFigure}
		\stepcounter{cntFigure}
	}
\end{wrapfigure}
}
