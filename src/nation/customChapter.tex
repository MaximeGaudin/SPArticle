\newcommand{\chapterfont}{
	\fontencoding{\encodingdefault}
	\fontfamily{pag}
	\fontseries{bc}
	\fontshape{bf}
	\selectfont}

\newcommand{\chapternumfont}{
	\fontencoding{\encodingdefault}
	\fontfamily{pag}
	\fontseries{bc}
	\fontshape{bf}
	\fontsize{32}{30}
	\selectfont
}

\newcounter{chapitre}
\setcounter{chapitre}{1}

\makeatletter
\renewcommand{\chapter}{%
	\cleardoublepage
	\stepcounter{chapitre}
	\thispagestyle{plain}
	\global\@topnum\z@
	\@afterindentfalse
	\secdef\@chapter\@schapter}

\newenvironment{cadrechap}
{
	\begin{list}{}{%
		\setlength{\leftmargin}{0pt}%
		\setlength{\rightmargin}{0cm}%
		\setlength{\itemindent}{0pt}%
		\setlength{\labelsep}{0pt}%
	}
	\item
}
{
	\end{list}
}

% Document
\renewcommand{\@makechapterhead}[1]{% 
		% Numéro de chapitre
		{\chapternumfont\thechapter}%
		% Titre du chapitre
		\hfill{\huge\chapterfont#1}%  

	\vspace*{1cm}
		\par\minitoc
	\vspace*{1cm}
}

% Table des matières
\renewcommand{\@makeschapterhead}[1]{%   
	\begin{cadrechap}
		\mbox{}\hfill{\bf\huge#1}
	\end{cadrechap}%
	\vspace{2cm}\par
}
\makeatother
