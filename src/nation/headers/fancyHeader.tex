\newcounter{cntPage}

\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\fancyhead[C]{
 	\setcounter{cntPage}{\value{page}}%
 	\addtocounter{cntPage}{-1}%
 	%
 	\ifthenelse{\not \equal{\thecntPage}{\pageref{LastPage}}}%
	{\sl\author}
	{\fancyhf{}\vspace*{0.3\textheight}\Huge\bf Fin\\ %
	\vspace*{0.1\textheight}\ifthenelse{\isundefined{\documentlogo}}{}{\image{\documentlogo}{0.4}}}%
}

\fancyfoot[C]{%
 	\setcounter{cntPage}{\value{page}}%
 	\addtocounter{cntPage}{-1}%
 	%
 	\ifthenelse{\not \equal{\thecntPage}{\pageref{LastPage}}}%
	{\thepage\ sur\ \pageref{LastPage}}%
	{\fancyhf{}}%
}

% Heading des pages de style plain
\fancypagestyle{plain}{%
	\fancyhf{}%
	\fancyfoot[C]{\thepage\ sur\ \pageref{LastPage}}%
	\renewcommand{\headrulewidth}{0pt}%
}
